# Framework web principal de alta performance para a construção da API
fastapi

# Servidor ASGI (Asynchronous Server Gateway Interface) para executar a aplicação FastAPI
uvicorn[standard]

# Biblioteca para realizar requisições HTTP (usada para consumir os dados externos do football-data.org)
requests

# ORM (Object-Relational Mapping) para interagir com o banco de dados usando classes Python em vez de SQL puro
sqlalchemy

# Driver do PostgreSQL para Python (a versão "binary" dispensa compilação, ideal para containers Docker leves)
psycopg2-binary

# Utilizado para cálculos matemáticos e estatísticos (ex: Distribuição de Poisson na simulação de Monte Carlo)
numpy

# Validação de dados e criação dos esquemas (schemas) de entrada e saída para a documentação automática do Swagger
pydantic
